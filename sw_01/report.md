### 2. Создание структуры базы данных

Создана база данных work со следующими таблицами:
- departments — отделы компании (8 отделов)
- positions — должности (10 должностей)
- employees — сотрудники (30 человек)
- employment — трудоустройство (зарплаты, даты начала работы)
- kpi — показатели эффективности (70+ записей за 2023-2024 гг.)

``` sql
CREATE TABLE departments (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    location VARCHAR(200)
);

CREATE TABLE positions (
    id SERIAL PRIMARY KEY,
    title VARCHAR(100) NOT NULL,
    description TEXT
);

CREATE TABLE employees (
    id SERIAL PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    hire_date DATE NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    birth_date DATE
);

CREATE TABLE employment (
    id SERIAL PRIMARY KEY,
    employee_id INT REFERENCES employees(id),
    department_id INT REFERENCES departments(id),
    position_id INT REFERENCES positions(id),
    salary NUMERIC(10,2) NOT NULL,
    start_date DATE NOT NULL DEFAULT CURRENT_DATE,
    end_date DATE
);

CREATE TABLE kpi (
    id SERIAL PRIMARY KEY,
    employee_id INT REFERENCES employees(id),
    year INT NOT NULL,
    month INT NOT NULL,
    kpi_score DECIMAL(5,2) CHECK (kpi_score >= 0 AND kpi_score <= 100),
    tasks_completed INT DEFAULT 0,
    projects_completed INT DEFAULT 0,
    client_satisfaction DECIMAL(3,2) CHECK (client_satisfaction >= 1 AND client_satisfaction <= 5),
    notes TEXT
);
  ```
